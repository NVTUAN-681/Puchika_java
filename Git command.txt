cd C:
cd Git

git config --global user.name "<user name>" : Khai báo user name
git config --global user.email "<user email>" : Khai báo user email
cat ~/.gitconfig : Kiểm tra thông tin user
git log: kiểm tra thông tin nhánh

mkdir <name folder> : Tạo 1 folder mới
cd <name folder> : Di chuyển đến 1 folder
git init : Tạo 1 local repository
git clone <path to repo on github> : Kéo folder trên github về máy mình

echo "file content" >> "filename"."file extension" : Tạo 1 file mới
git add <file name> : Thêm file vào repo
git commit -m "<description>" : Mô tả, xác nhận cho hành động thêm
git add . : thêm toàn bộ file trong folder

git branch -M main: Tạo nhánh chính
git remote add origin <path to repo on github> : Khai báo <đường dẫn> lên github, VD: <https://github.com/Blue-cloud-droid/Hello-world.git>
git branch <branch name> : Tạo nhánh phụ
git checkout <branch name> : Chuyển nhánh
git checkout -b <branch name> : Tạo và chuyển nhánh
git push --set-upstream origin <branch name>: Tạo nhánh mới trên git
git branch --set-upstream-to=origin/<branch> master : Kết nối với nhánh <branch> trên git để pull, fork, clone, push

git push: Đẩy tất cả local repo lên github
git push -u origin main: Đẩy file từ local repo lên github nhánh chính
git push -u origin <branch name>: Đẩy file từ local repo lên 1 nhánh của github (main cũng là 1 nhánh)
git pull: Kéo, đồng bộ file từ github về máy mình
git pull origin <branch name>: Đồng bộ nhánh được gọi tên
git merge <branch name>: gộp nhánh phụ được gọi tên về nhánh đang ở (có thể cần đồng bộ nhánh đang ở với github), (sau khi gửi lệnh merge lên github có thể cần "Pull request" và "Accept request")
git merge origin/<branch name> --allow-unrelated-histories: hợp nhất 2 nhánh không chung nguồn
